  def call_action(action, record)
    action = @options[action]
    catch :halt_aasm_chain do
      action.is_a?(Array) ?
              action.each {|a| _call_action(a, record)} :
              _call_action(action, record)
    end
  end

  def update(options = {})
    if options.key?(:display) then
      @display_name = options.delete(:display)
    end
    @options = options
    self
  end
